const addRow = document.getElementById('add-row');
const removeRow = document.getElementById('remove-row');
const addColumn = document.getElementById('add-column');
const removeColumn = document.getElementById('remove-column');
const grid = document.getElementById('grid');


console.log(grid);
addRow.addEventListener('click', () => {
    console.log(grid.children[0]);
    for (let i = 0; i < grid.children.length; i++) {
        const newCell = document.createElement("div");
        newCell.className = "item";
        grid.children[i].appendChild(newCell);
    }
})

removeRow.addEventListener('click', () => {
    const children = grid.children;
    for (let i = 0; i < children.length; i++) {
        const column = children[i];
        const columnChildren = column.children;
        console.log(column.children);
        const rowToRemove = columnChildren[columnChildren.length-1];
        console.log(rowToRemove);
        column.removeChild(rowToRemove);
    }
})

addColumn.addEventListener('click', () => {
    const newColumn = grid.children[0];
    console.log(newColumn);
    grid.appendChild(newColumn);
})
